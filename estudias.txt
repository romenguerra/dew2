 
// VARIABLES GLOBALES
let pantalla = document.getElementById("pantalla");
let numeroActual = "0";
let numeroAnterior = "";
let operador = null;
let reiniciarPantalla = false;

// OBTENER TODOS LOS BOTONES
const botones = document.querySelectorAll("button");

// ASIGNAR EVENTO A CADA BOTÓN
botones.forEach((boton) => {
    boton.addEventListener("click", function () {
        const valor = this.textContent;

        // VER QUÉ TIPO DE BOTÓN ES
        if (this.classList.contains("numero")) {
            ingresarNumero(valor);
        } else if (this.classList.contains("operador")) {
            ingresarOperador(valor);
        } else if (this.classList.contains("especial")) {
            ejecutarEspecial(valor);
        }

        actualizarPantalla();
    });
});

// FUNCIÓN PARA INGRESAR NÚMEROS
function ingresarNumero(numero) {
    if (numeroActual === "0" || reiniciarPantalla) {
        numeroActual = numero;
        reiniciarPantalla = false;
    } else {
        // Evitar múltiples puntos decimales
        if (numero === "." && numeroActual.includes(".")) {
            return;
        }
        numeroActual += numero;
    }
}

// FUNCIÓN PARA INGRESAR OPERADORES
function ingresarOperador(nuevoOperador) {
    // Si ya hay un operador pendiente, calcular primero
    if (operador !== null && !reiniciarPantalla) {
        calcular();
    }

    // Guardar el número actual como anterior
    numeroAnterior = numeroActual;
    operador = nuevoOperador;
    reiniciarPantalla = true;
}

// FUNCIÓN PARA BOTONES ESPECIALES
function ejecutarEspecial(valor) {
    switch (valor) {
        case "C": // Clear - Limpiar todo
            numeroActual = "0";
            numeroAnterior = "";
            operador = null;
            reiniciarPantalla = false;
            break;

        case "⌫": // Borrar último dígito
            if (numeroActual.length > 1) {
                numeroActual = numeroActual.slice(0, -1);
            } else {
                numeroActual = "0";
            }
            break;

        case "%": // Porcentaje
            numeroActual = (parseFloat(numeroActual) / 100).toString();
            break;
    }
}

// FUNCIÓN PARA CALCULAR
function calcular() {
    if (operador === null || numeroAnterior === "") {
        return;
    }

    const num1 = parseFloat(numeroAnterior);
    const num2 = parseFloat(numeroActual);

    if (isNaN(num1) || isNaN(num2)) {
        return;
    }

    let resultado;

    switch (operador) {
        case "+":
            resultado = num1 + num2;
            break;
        case "−":
            resultado = num1 - num2;
            break;
        case "×":
            resultado = num1 * num2;
            break;
        case "÷":
            if (num2 === 0) {
                resultado = "Error";
            } else {
                resultado = num1 / num2;
            }
            break;
        default:
            return;
    }

    // Limitar decimales
    if (typeof resultado === "number") {
        // Si es entero, mostrar sin decimales
        // Si tiene decimales, mostrar máximo 10 decimales
        resultado = resultado.toString();
        if (resultado.includes(".")) {
            resultado = parseFloat(resultado).toFixed(10);
            // Eliminar ceros innecesarios al final
            resultado = parseFloat(resultado).toString();
        }
    }

    numeroActual = resultado.toString();
    operador = null;
    numeroAnterior = "";
    reiniciarPantalla = true;
}

// FUNCIÓN PARA ACTUALIZAR LA PANTALLA
function actualizarPantalla() {
    pantalla.value = numeroActual;

    // Limitar longitud de la pantalla
    if (numeroActual.length > 12) {
        pantalla.style.fontSize = "1.5em";
    } else {
        pantalla.style.fontSize = "2em";
    }
}

// EVENTO PARA EL BOTÓN IGUAL
document.getElementById("igual").addEventListener("click", function () {
    calcular();
    actualizarPantalla();
});

// EVENTOS DE TECLADO (EXTRA PARA EL EXAMEN)
document.addEventListener("keydown", function (evento) {
    const tecla = evento.key;

    if (tecla >= "0" && tecla <= "9") {
        ingresarNumero(tecla);
    } else if (tecla === ".") {
        ingresarNumero(tecla);
    } else if (tecla === "+" || tecla === "-") {
        ingresarOperador(tecla);
    } else if (tecla === "*") {
        ingresarOperador("×");
    } else if (tecla === "/") {
        evento.preventDefault(); // Evitar que se abra quick find
        ingresarOperador("÷");
    } else if (tecla === "Enter" || tecla === "=") {
        evento.preventDefault();
        calcular();
    } else if (tecla === "Escape") {
        ejecutarEspecial("C");
    } else if (tecla === "Backspace") {
        ejecutarEspecial("⌫");
    }

    actualizarPantalla();
});

// INICIALIZAR
actualizarPantalla();
